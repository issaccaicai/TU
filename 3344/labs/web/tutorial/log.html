Step 5: Javascript

- Last, we have the javascript for this filtering gallery.

  • This filterSelection function will display and show all columns images. All the filtered elements will be displayed. Removed any elements that are not selected.
  • We assgin filterSelection function with id "all", which means we are implementing the function, and select all images to be displayed as default. 
                    filterSelection("all")
                    function filterSelection(img) 
                        var x, i; // declare vairable x, and i
                        x = document.getElementsByClassName("column");
                        if (img == "all")
                            img = "";
                              // Add the "show" class to the filtered 
                                    elements, and remove the "show" class 
                                        from the elements that are not selected.
                        for (i = 0; i < x.length; i++) {
                            removeClass(x[i], "show");
                            if (x[i].className.indexOf(img) > -1)
                            addClass(x[i], "show");
                        }
                    }
  • This addClass function will display and show selected column images.
                    // Show filtered elements
                    function addClass(element, name) {
                        var i, arr1, arr2;
                        arr1 = element.className.split(" ");
                        arr2 = name.split(" ");
                        for (i = 0; i < arr2.length; i++) {
                            if (arr1.indexOf(arr2[i]) == -1) {
                                element.className += " " + arr2[i];
                            }
                        }
                    }
  • This remove function will hide any other elements which are not in selected column images.
                    // Hide elements that are not selected
                    function removeClass(element, name) {
                        var i, arr1, arr2;
                        arr1 = element.className.split(" ");
                        arr2 = name.split(" ");
                        for (i = 0; i < arr2.length; i++) {
                            while (arr1.indexOf(arr2[i]) > -1) {
                                arr1.splice(arr1.indexOf(arr2[i]), 1); 
                                   // splice 1 elements at arr1.indexOf(arr2[i])
                            }
                        }
                        element.className = arr1.join(" ");// concatenating all of the elements
                    }
